--- PHẦN 3: HÀM SANITIZE (chống prompt injection) ---

Thêm hàm này vào file, gọi trước khi đưa text vào prompt:

function sanitize(str) {
  if (!str) return '';
  return str
    .slice(0, 300)
    .replace(/[`'"\\]/g, '')
    .replace(/\n{3,}/g, '\n\n')
    .replace(/ignore|system|instruction|override|jailbreak/gi, '[x]')
    .trim();
}

--- PHẦN 4: XỬ LÝ RESPONSE GEMINI AN TOÀN ---

Thay thế code xử lý response hiện tại bằng:

async function callGeminiValuation(data) {
  let attempt = 0;
  while (attempt < 2) {  // thử tối đa 2 lần
    attempt++;
    try {
      const result = await model.generateContent(userPrompt);
      const rawText = result.response.text().trim();

      // Parse JSON an toàn
      let parsed;
      try {
        parsed = JSON.parse(rawText);
      } catch {
        // Gemini đôi khi wrap trong ```json ... ``` dù đã set mimeType
        const match = rawText.match(/\{[\s\S]*\}/);
        if (!match) throw new Error('Gemini không trả về JSON hợp lệ');
        parsed = JSON.parse(match[0]);
      }

      // Validate dữ liệu trả về
      if (
        typeof parsed.gia_uoc_tinh !== 'number' ||
        parsed.gia_uoc_tinh <= 0 ||
        parsed.gia_uoc_tinh > 1000 ||
        typeof parsed.do_tin_cay !== 'number' ||
        parsed.do_tin_cay < 50 ||
        parsed.do_tin_cay > 99 ||
        !parsed.breakdown
      ) {
        if (attempt < 2) continue;  // thử lại lần 2
        throw new Error('Dữ liệu định giá không hợp lệ');
      }

      return { success: true, data: parsed };

    } catch (err) {
      if (attempt >= 2) {
        return {
          success: false,
          error: 'Hệ thống AI tạm thời gặp sự cố. Vui lòng thử lại sau ít phút.'
        };
      }
    }
  }
}

--- PHẦN 5: RATE LIMITING ĐƠN GIẢN (không cần Redis) ---

Thêm vào file backend/API handler:

const _rl = new Map();
function checkLimit(ip) {
  const now = Date.now();
  const r = _rl.get(ip);
  if (!r || now > r.reset) {
    _rl.set(ip, { count: 1, reset: now + 86400000 });
    return true;
  }
  if (r.count >= 30) return false;
  r.count++;
  return true;
}

// Dùng trong handler:
const ip = req.headers['x-forwarded-for'] || req.ip || 'unknown';
if (!checkLimit(String(ip).split(',')[0].trim())) {
  return res.status(429).json({
    success: false,
    error: 'Bạn đã định giá quá 30 lần hôm nay. Thử lại sau 24 giờ.'
  });
}
