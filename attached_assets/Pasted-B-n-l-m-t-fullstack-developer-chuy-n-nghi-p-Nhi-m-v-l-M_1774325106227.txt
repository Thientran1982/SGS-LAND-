-- Bảng 1: Phòng ban
CREATE TABLE departments (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  name        TEXT NOT NULL,  
  description TEXT,
  created_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 2: Nhân viên
CREATE TABLE users (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  full_name     TEXT NOT NULL,
  email         TEXT UNIQUE NOT NULL,
  phone         TEXT,
  role          TEXT CHECK(role IN ('admin','manager','employee')) DEFAULT 'employee',
  department_id INTEGER REFERENCES departments(id),
  position      TEXT,           -- chức danh: "Trưởng phòng KD", "Chuyên viên PL"...
  is_active     INTEGER DEFAULT 1,
  created_at    DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 3: Dự án BĐS
CREATE TABLE projects (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  name          TEXT NOT NULL,
  code          TEXT UNIQUE,    -- mã dự án: SGS-HCM-01
  description   TEXT,
  property_type TEXT CHECK(property_type IN 
                  ('apartment','villa','shophouse','land','office','resort')),
  location      TEXT,           -- địa chỉ / khu vực
  status        TEXT CHECK(status IN 
                  ('planning','active','on_hold','completed','cancelled'))
                DEFAULT 'planning',
  manager_id    INTEGER REFERENCES users(id),
  start_date    DATE,
  end_date      DATE,
  created_at    DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 4: Công việc (Task)
CREATE TABLE tasks (
  id               INTEGER PRIMARY KEY AUTOINCREMENT,
  title            TEXT NOT NULL,
  description      TEXT,
  project_id       INTEGER REFERENCES projects(id),
  department_id    INTEGER REFERENCES departments(id),
  category         TEXT CHECK(category IN (
                     'sales',        -- bán hàng
                     'legal',        -- pháp lý
                     'marketing',    -- marketing
                     'site_visit',   -- khảo sát thực địa
                     'customer_care',-- chăm sóc khách hàng
                     'finance',      -- tài chính / thanh toán
                     'construction', -- xây dựng / kỹ thuật
                     'admin',        -- hành chính
                     'other'
                   )),
  status           TEXT CHECK(status IN (
                     'todo',         -- chờ làm
                     'in_progress',  -- đang thực hiện
                     'review',       -- chờ duyệt
                     'done',         -- hoàn thành
                     'cancelled'     -- đã huỷ
                   )) DEFAULT 'todo',
  priority         TEXT CHECK(priority IN ('low','medium','high','urgent')) 
                   DEFAULT 'medium',
  deadline         DATE,
  estimated_hours  REAL,            -- giờ công ước tính
  actual_hours     REAL,            -- giờ công thực tế (cập nhật khi done)
  completion_note  TEXT,            -- ghi chú khi đánh dấu hoàn thành
  created_by       INTEGER REFERENCES users(id),
  created_at       DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at       DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 5: Phân công công việc
CREATE TABLE task_assignments (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  task_id     INTEGER NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
  user_id     INTEGER NOT NULL REFERENCES users(id),
  assigned_by INTEGER REFERENCES users(id),
  assigned_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  due_note    TEXT,                 -- ghi chú khi giao
  is_primary  INTEGER DEFAULT 0,   -- 1 = người chịu trách nhiệm chính
  UNIQUE(task_id, user_id)
);

-- Bảng 6: Bình luận / trao đổi trong task
CREATE TABLE task_comments (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  task_id    INTEGER NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
  user_id    INTEGER NOT NULL REFERENCES users(id),
  content    TEXT NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 7: Lịch sử thay đổi
CREATE TABLE activity_logs (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  task_id    INTEGER REFERENCES tasks(id) ON DELETE CASCADE,
  user_id    INTEGER REFERENCES users(id),
  action     TEXT NOT NULL,   -- 'created','status_changed','assigned',
                              -- 'unassigned','commented','updated','deleted'
  old_value  TEXT,            -- JSON string giá trị cũ
  new_value  TEXT,            -- JSON string giá trị mới
  detail     TEXT,            -- mô tả ngắn dạng text
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Bảng 8: Nhắc việc / deadline reminder
CREATE TABLE task_reminders (
  id           INTEGER PRIMARY KEY AUTOINCREMENT,
  task_id      INTEGER NOT NULL REFERENCES tasks(id) ON DELETE CASCADE,
  remind_date  DATE NOT NULL,
  remind_note  TEXT,
  is_sent      INTEGER DEFAULT 0,
  created_by   INTEGER REFERENCES users(id),
  created_at   DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Trigger: tự động cập nhật updated_at khi task thay đổi
CREATE TRIGGER update_task_timestamp
AFTER UPDATE ON tasks
BEGIN
  UPDATE tasks SET updated_at = CURRENT_TIMESTAMP WHERE id = NEW.id;
END;

BACKEND — BUSINESS LOGIC CHI TIẾT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

=== TASK LOGIC ===

1. TẠO TASK (POST /api/tasks)
   Validate:
   - title: bắt buộc, 5-200 ký tự
   - deadline: không được là ngày trong quá khứ
   - project_id: phải tồn tại nếu có
   - estimated_hours: số dương nếu có
   Sau khi tạo: ghi activity_log action='created'

2. CẬP NHẬT STATUS (PATCH /api/tasks/:id/status)
   Logic chuyển trạng thái hợp lệ:
     todo        → in_progress, cancelled
     in_progress → review, todo, cancelled
     review      → done, in_progress
     done        → (không cho quay lại, chỉ admin mới đổi)
     cancelled   → todo (reopen, chỉ admin/manager)
   Khi chuyển sang done:
   - Lưu actual_hours nếu được cung cấp
   - Lưu completion_note nếu có
   - Ghi activity_log
   Khi chuyển sang cancelled:
   - Bắt buộc có lý do (ghi vào completion_note)

3. GIAO VIỆC (POST /api/tasks/:id/assign)
   Body: { user_ids: number[], due_note?: string, primary_user_id?: number }
   Logic:
   - Mỗi task có thể giao cho nhiều người
   - Nhưng chỉ có 1 người chịu trách nhiệm chính (is_primary=1)
   - Nếu task đang ở status 'todo' → tự động chuyển 'in_progress'
   - Không được giao cho user is_active=0
   - Ghi activity_log cho từng người được giao
   - Trả về danh sách assignees đầy đủ sau khi giao

4. HỦY GIAO VIỆC (DELETE /api/tasks/:id/assign/:userId)
   Logic:
   - Nếu huỷ người is_primary, yêu cầu chỉ định người primary mới
     hoặc task không còn ai → trả về lỗi 400 với message rõ ràng
   - Ghi activity_log action='unassigned'

5. TÍNH TRẠNG THÁI QUÁ HẠN:
   Mọi API trả về task đều tính thêm field:
   - is_overdue: boolean  (deadline < hôm nay AND status != 'done'/'cancelled')
   - days_until_deadline: number (âm = quá hạn, 0 = hôm nay, dương = còn hạn)
   - urgency_level: 'overdue' | 'critical' | 'warning' | 'normal'
     + overdue  : quá hạn
     + critical : còn ≤ 1 ngày
     + warning  : còn ≤ 3 ngày
     + normal   : còn > 3 ngày

=== PROJECT LOGIC ===

6. TỔNG HỢP TIẾN ĐỘ DỰ ÁN:
   Mỗi project response kèm:
   - total_tasks, todo_count, in_progress_count, done_count, overdue_count
   - completion_rate: (done / total) * 100
   - on_schedule: true nếu không có task quá hạn

=== USER WORKLOAD LOGIC ===

7. WORKLOAD NHÂN VIÊN:
   GET /api/users/:id/workload
   Trả về:
   - active_tasks: số tasks đang in_progress
   - overdue_tasks: số tasks quá hạn chưa xong
   - completed_this_week: số tasks done trong 7 ngày qua
   - completed_this_month: số tasks done trong 30 ngày qua
   - avg_completion_days: trung bình số ngày hoàn thành task
   - workload_score: tính theo công thức:
       (active_tasks * 1) + (overdue_tasks * 2) + 
       (urgent_tasks * 1.5) → dùng để sort "bận nhất"

=== DASHBOARD STATS ===

8. GET /api/dashboard/stats
   Trả về JSON:
   {
     overview: {
       total_tasks, todo, in_progress, review, done, cancelled,
       overdue_count, due_today_count, due_this_week_count
     },
     completion_rate_today: số tasks done hôm nay,
     completion_rate_week:  số tasks done tuần này,
     by_priority: { urgent, high, medium, low },
     by_category: { sales, legal, marketing, ... },
     by_project:  [{ project_id, name, total, done, overdue }],
     top_overdue_tasks: [5 tasks quá hạn lâu nhất, kèm assignees],
     upcoming_deadlines: [10 tasks sắp đến hạn trong 7 ngày],
     workload_by_user: [
       { user_id, name, department, active_tasks, overdue_tasks, workload_score }
       -- sort theo workload_score giảm dần
     ]
   }

=== SEARCH & FILTER LOGIC ===

9. GET /api/tasks — Query parameters đầy đủ:
   ?status=        (có thể nhiều: ?status=todo,in_progress)
   ?priority=      (có thể nhiều)
   ?project_id=
   ?department_id=
   ?category=
   ?assignee_id=   (lọc task có người này trong assignees)
   ?created_by=
   ?is_overdue=true/false
   ?deadline_from= (YYYY-MM-DD)
   ?deadline_to=
   ?search=        (tìm trong title + description)
   ?sort_by=       (deadline|priority|created_at|updated_at)
   ?sort_dir=      (asc|desc)
   ?page=          (default 1)
   ?limit=         (default 20, max 100)

   Response:
   {
     data: Task[],
     pagination: { page, limit, total, total_pages },
     filters_applied: { ... }  -- echo lại filter đang dùng
   }

=== REPORT LOGIC ===

10. GET /api/reports/summary
    Bảng tổng hợp nhân viên:
    - Group by user
    - Mỗi row: tên, phòng ban, todo, in_progress, done, 
      overdue, total_assigned, completion_rate, avg_days
    - Sort: overdue giảm dần (người có task trễ nhiều lên trên)

11. GET /api/reports/export/csv
    Tạo và trả về file CSV với headers:
    ID, Tiêu đề, Dự án, Danh mục, Ưu tiên, Trạng thái,
    Người thực hiện, Người giao, Deadline, 
    Ngày hoàn thành, Giờ ước tính, Giờ thực tế, Ghi chú
    Content-Type: text/csv; charset=utf-8
    Filename: sgs-tasks-export-{YYYYMMDD}.csv

=== ERROR HANDLING ===

12. Mọi route đều có:
    - Try/catch với error middleware tập trung
    - Validate input bằng Zod trước khi xử lý
    - HTTP status code chuẩn:
      200 OK | 201 Created | 400 Bad Request | 
      404 Not Found | 409 Conflict | 422 Unprocessable | 500 Server Error
    - Response lỗi chuẩn: { error: true, code: string, message: string, details?: any }

    Đang làm | Hoàn thành tháng này | Quá hạn | Trung bình ngày/task
  - Danh sách tasks đang giao:
    Có filter: Status / Priority / Dự án
    Mỗi task: tiêu đề / deadline / priority / status
    Click task → navigate đến /tasks/:id
  - Biểu đồ mini: số tasks done theo tuần (6 tuần gần nhất)

=== TRANG DỰ ÁN (/projects) ===

Danh sách card (3 cột):
  Mỗi card: mã dự án / tên / loại BĐS / địa điểm / 
  Trưởng dự án / Status / Progress bar (% done) /
  Số tasks tổng | done | overdue

Click vào dự án → trang chi tiết:
  - Thông tin dự án đầy đủ
  - Kanban mini chỉ hiện tasks của dự án này
  - Danh sách thành viên có giao task trong dự án
  - Nút "Tạo task cho dự án này" (pre-fill project)

=== TRANG BÁO CÁO (/reports) ===

Tab 1 — THEO NHÂN VIÊN:
  Bảng: Tên | Phòng ban | Chờ làm | Đang làm | Hoàn thành | 
        Quá hạn | Tổng | Tỉ lệ HT% | TB ngày/task
  Logic highlight: hàng có overdue > 0 → nền khác biệt
  Sort mặc định: overdue giảm dần

Tab 2 — THEO DỰ ÁN:
  Bảng: Tên DA | Loại | Tổng tasks | Hoàn thành | Đang làm | 
        Chờ làm | Quá hạn | Tiến độ%
  Progress bar trực quan cho từng dự án

Tab 3 — THEO THỜI GIAN:
  Date range picker: chọn khoảng thời gian
  Biểu đồ cột nhóm: tasks tạo mới vs hoàn thành (theo tuần)
  Biểu đồ đường: tích lũy tasks done theo ngày

Nút Export:
  - Export CSV toàn bộ hoặc theo filter hiện tại
  - In báo cáo (CSS print-friendly, ẩn sidebar/header)
